Properties

ParameterTypeDefaultDescription
query String required Current search query text
onQueryChange (String) -> Unit required Callback when query text changes
onSearch (String) -> Unit required Callback when search is submitted
isExpandable Boolean true Whether the search bar can collapse/expand
isExpanded Boolean false Current expansion state
placeholder @Composable null Placeholder content when query is empty
leadingIconOnCollapse @Composable null Leading icon shown in collapsed state (e.g., search icon)

Usage

@Composable
fun SearchScreen() {
    var query by remember { mutableStateOf("") }
    var isExpanded by rememberSaveable { mutableStateOf(false) }

    EpisodiveSearchBar(
        query = query,
        onQueryChange = { query = it },
        onSearch = { /* submit search */ },
        isExpandable = true,
        isExpanded = isExpanded,
        placeholder = {
            Text("Search podcasts...")
        },
        leadingIconOnCollapse = {
            Icon(TablerIcons.Search, contentDescription = "Search")
        },
        leadingIconOnExpand = {
            IconButton(onClick = { isExpanded = false }) {
                Icon(TablerIcons.ArrowLeft, contentDescription = "Back")
            }
        },
        trailingIcon = {
            if (query.isNotEmpty()) {
                IconButton(onClick = { query = "" }) {
                    Icon(TablerIcons.X, contentDescription = "Clear")
                }
            }
        }
    )
}
