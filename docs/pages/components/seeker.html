Usage

@Composable
fun PlayerControls(
    progress: Progress,
    chapters: List<Chapter>,
    onSeekTo: (Long) -> Unit
) {
    EpisodiveSeeker(
        progress = Progress(
            position = 768000L,   // 12:48
            buffered = 1170000L,  // 19:30
            duration = 1920000L   // 32:00
        ),
        onSeekTo = onSeekTo,
        chapters = chapters,
        isControllable = true
    )
}

// Chapter data class
data class Chapter(
    val title: String,
    val startTime: Long,  // ms
    val endTime: Long     // ms
)
