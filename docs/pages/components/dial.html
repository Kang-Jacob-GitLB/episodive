Usage

@Composable
fun PlaybackSpeedControl(
    currentSpeed: Float,
    onSpeedChange: (Float) -> Unit
) {
    Column(horizontalAlignment = Alignment.CenterHorizontally) {
        Text(
            text = "${String.format("%.1f", currentSpeed)}x",
            style = MaterialTheme.typography.headlineMedium
        )

        EpisodiveDial(
            value = currentSpeed,
            onValueChange = onSpeedChange,
            range = 0.5f..3.5f,
            steps = 30,
            selectedColor = MaterialTheme.colorScheme.primary,
            unselectedColor = MaterialTheme.colorScheme.outline
        )
    }
}

// Features:
// - VelocityTracker for momentum-based fling
// - splineBasedDecay for natural deceleration
// - spring animation for snapping to nearest step
// - Horizontal drag gesture with pointer input

Design Specs
