DominantRegion

The Palette API extracts dominant colors from a specific region of the image. The dominantRegion parameter controls which area is sampled.

Usage

@Composable
fun PodcastArtwork(podcast: Podcast) {
    var dominantColor by remember { mutableStateOf(Color.Transparent) }

    StateImage(
        imageUrl = podcast.imageUrl,
        size = 300,
        dominantRegion = DominantRegion.Bottom,
        onDominantColorExtracted = { color ->
            dominantColor = color
        },
        fallbackIcon = TablerIcons.Microphone,
        modifier = Modifier
            .size(150.dp)
            .clip(RoundedCornerShape(12.dp))
    )

    // Use dominantColor for gradient background, etc.
    Box(
        modifier = Modifier.background(
            Brush.verticalGradient(
                colors = listOf(dominantColor, Color.Transparent)
            )
        )
    )
}
